## Overview

SMyEG implements a comprehensive **5-level hierarchical system** for managing forest and land patrimony. This structure enables detailed tracking from large land parcels down to individual measurement plots, supporting biological asset management, inventory control, and geospatial operations.

## Hierarchy Overview

<Steps>
  <Step title="Level 2 - Land Parcels">
    Large land units: FINCA, PREDIO, HATO, FUNDO, HACIENDA, ABRAE
  </Step>
  <Step title="Level 3 - Management Sections">
    Divisions: COMPARTIMIENTO, BLOCK, SECCION, LOTE, ZONA, ZONIFICACION
  </Step>
  <Step title="Level 4 - Management Units">
    Operational units: RODAL, PARCELA, ENUMERATION, UNIDAD_DE_MANEJO, CONUCO
  </Step>
  <Step title="Level 5 - Measurement Plots">
    Sampling units: REFERENCIA, SUBUNIDAD, SUBPARCELA, MUESTRA
  </Step>
  <Step title="Level 6 - Biological Assets">
    Individual tree stands and inventory records
  </Step>
</Steps>

### Land Use Tracking

Level 4 tracks current and historical land use:

- **currentLandUseName**: Current land use classification
- **previousLandUseName**: Previous use (for change detection)
- **landUseChangeDate**: When the land use changed
- **rotationPhase**: Current growth/harvest cycle phase

<Warning>
  Level 4 is the **primary unit** for geospatial operations, biological asset tracking, and measurement campaigns.
</Warning>

### Plot Area Calculation

The system automatically calculates `areaM2` based on shape and dimensions:

```typescript
// From src/lib/forest-area.ts
export function calculatePlotAreaM2(plot: {
  shapeType: PlotShapeType;
  dimension1M: number | null;
  dimension2M: number | null;
  dimension3M?: number | null;
  dimension4M?: number | null;
}): number | null {
  switch (plot.shapeType) {
    case "RECTANGULAR":
      return (dimension1M ?? 0) * (dimension2M ?? 0);
    
    case "CUADRADA":
      return (dimension1M ?? 0) ** 2;
    
    case "CIRCULAR":
      const radius = (dimension1M ?? 0) / 2;
      return Math.PI * radius ** 2;
    
    case "HEXAGONAL":
      const side = dimension1M ?? 0;
      return (3 * Math.sqrt(3) / 2) * side ** 2;
  }
}
```

## API Routes

### Forest Patrimony CRUD

```typescript
// GET /api/forest/patrimony?level=2&page=1&limit=25
export async function GET(req: NextRequest) {
  const authResult = await requireAuth();
  if ("error" in authResult) return authResult.error;
  
  const { level, page, limit, parentId, search } = queryResult.data;
  
  // Organization scoping
  const organizationId = await resolveOrganizationId({
    id: authResult.session.user.id,
    organizationId: authResult.session.user.organizationId,
  });
  
  if (level === "2") {
    const where: Prisma.ForestPatrimonyLevel2WhereInput = {
      ...(isOrgScoped ? { organizationId } : {}),
      ...(search ? {
        OR: [
          { code: { contains: search, mode: "insensitive" } },
          { name: { contains: search, mode: "insensitive" } },
        ],
      } : {}),
    };
    
    const [total, items] = await Promise.all([
      prisma.forestPatrimonyLevel2.count({ where }),
      prisma.forestPatrimonyLevel2.findMany({
        where,
        skip: (page - 1) * limit,
        take: limit,
        orderBy: [{ createdAt: "desc" }],
      }),
    ]);
    
    return ok({ items, pagination: { total, page, limit } });
  }
  
  // Similar logic for levels 3, 4, 5
}
```

### Update Level 4 (with Geometry Management)

```typescript
// PATCH /api/forest/patrimony
export async function PATCH(req: NextRequest) {
  const updated = await prisma.$transaction(async (tx) => {
    const updatedLevel4 = await tx.forestPatrimonyLevel4.update({
      where: { id: parsed.data.id },
      data: level4Data,
    });
    
    // Deactivate geometries when Level 4 is deactivated
    if (level4Data.isActive === false && current.isActive) {
      await tx.$executeRaw`
        UPDATE public.forest_geometry_n4
        SET is_active = FALSE,
            valid_to = COALESCE(valid_to, NOW()),
            updated_at = NOW()
        WHERE organization_id = ${organizationId}::uuid
          AND level4_id = ${parsed.data.id}::uuid
          AND is_active = TRUE
      `;
    }
    
    return updatedLevel4;
  });
  
  // Enqueue geometry recalculation job
  if (organizationId) {
    void enqueueLevel4RecalcJob({
      organizationId,
      level4Id: updated.id,
      createdById: authResult.session.user.id,
    });
  }
  
  return ok(updated);
}
```

## Hierarchical Queries

Fetch complete hierarchy for a Level 4 unit:

```typescript
const level4 = await prisma.forestPatrimonyLevel4.findUnique({
  where: { id },
  include: {
    level3: {
      include: {
        level2: {
          include: {
            organization: true,
            neighbors: true,
          },
        },
      },
    },
    level5Units: true,
    level6Assets: true,
    geometryVersions: {
      where: { isActive: true },
      orderBy: { validFrom: 'desc' },
```
