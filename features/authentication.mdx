## Overview

SMyEG implements a robust authentication system using **NextAuth.js v5** with JWT-based sessions, multi-organization support, and enterprise security features including account lockouts, audit logging, and password reset flows.

## Authentication Architecture

### NextAuth v5 Configuration

The authentication system is configured in `src/lib/auth.ts`:

```typescript
import NextAuth from "next-auth";
import Credentials from "next-auth/providers/credentials";

export const { handlers, auth, signIn, signOut } = NextAuth({
  trustHost: true,
  session: { strategy: "jwt" },
  pages: { signIn: "/login" },
  providers: [
    Credentials({
      async authorize(credentials) {
        // Custom authentication logic
      }
    })
  ],
  callbacks: {
    async jwt({ token, user }) {
      // Enrich JWT with roles, permissions, organization
    },
    async session({ session, token }) {
      // Map token data to session
    }
  }
});
```

## User Model & Status Management

### User Schema (Prisma)

```prisma
model User {
  id                   String        @id @default(uuid())
  organizationId       String?       // Multi-tenant association
  email                String        @unique
  emailVerifiedAt      DateTime?
  username             String?       @unique
  passwordHash         String?
  authProvider         AuthProvider  @default(LOCAL)
  providerId           String?
  status               UserStatus    @default(PENDING_VERIFICATION)
  
  // Security features
  failedLoginAttempts  Int          @default(0)
  lockedUntil          DateTime?
  mfaEnabled           Boolean      @default(false)
  mfaSecret            String?
  mfaBackupCodes       String[]
  
  // Profile
  firstName            String?
  lastName             String?
  displayName          String?
  avatarUrl            String?
  
  // Relations
  organization         Organization?
  userRoles            UserRole[]
  sessions             Session[]
  auditLogs            AuditLog[]
}

enum UserStatus {
  ACTIVE
  INACTIVE
  PENDING_VERIFICATION
  LOCKED
  DELETED
}

enum AuthProvider {
  LOCAL
  GOOGLE
  GITHUB
  MICROSOFT
}
```

## Authentication Flow

### Login Process

<Steps>
  <Step title="User submits credentials">
    Username/email, password, and **organizationId** are required
  </Step>
  <Step title="User validation">
    System checks:
    - User exists
    - User status is ACTIVE
    - Account is not locked
    - Password is valid
  </Step>
  <Step title="Organization authorization">
    Validates user has access to the selected organization (see [Organizations](/features/organizations))
  </Step>
  <Step title="Load permissions">
    Fetches user roles and permissions via `getUserRolesAndPermissions()`
  </Step>
  <Step title="Create session">
    JWT token is created with user ID, roles, permissions, and organization context
  </Step>
  <Step title="Audit logging">
    LOGIN event is recorded in AuditLog table
  </Step>
</Steps>

### Failed Login Handling

```typescript
// From src/lib/auth.ts
if (!valid) {
  const attempts = user.failedLoginAttempts + 1;
  await prisma.user.update({
    where: { id: user.id },
    data: {
      failedLoginAttempts: attempts,
      lockedUntil: attempts >= 5 
        ? new Date(Date.now() + 30 * 60 * 1000) // 30 min lockout
        : null,
    },
  });

  await prisma.auditLog.create({
    data: {
      userId: user.id,
      action: "LOGIN_FAILED",
      entityType: "User",
      entityId: user.id,
    },
  });

  return null;
}
```

<Warning>
  After **5 failed login attempts**, the account is locked for **30 minutes**. The `lockedUntil` timestamp prevents authentication during this period.
</Warning>

## Session Management

### JWT Session Structure

```typescript
// Session includes:
interface Session {
  user: {
    id: string;
    email: string;
    name: string;
    roles: string[];              // ["USER", "ADMIN", etc.]
    permissions: string[];        // ["forest-patrimony:READ", ...]
    organizationId: string | null;
    organizationName: string | null;
  }
}
```

### Session Table (Database)

```prisma
model Session {
  id           String    @id @default(uuid())
  userId       String
  tokenHash    String    @unique
  refreshToken String?   @unique
  ipAddress    String?
  userAgent    String?
  deviceInfo   Json?
  isActive     Boolean   @default(true)
  expiresAt    DateTime
  lastActivity DateTime  @default(now())
  createdAt    DateTime  @default(now())
  revokedAt    DateTime?
  
  user         User      @relation(fields: [userId], references: [id])
  auditLogs    AuditLog[]
}
```

## User Registration

### Registration Endpoint

```typescript
// POST /api/auth/register
import { hashPassword } from "@/lib/crypto";
import { ensureRoleWithPermissions } from "@/lib/role-provisioning";

const passwordHash = await hashPassword(password);

const user = await prisma.user.create({
  data: {
    email,
    firstName,
    lastName,
    passwordHash,
    status: "PENDING_VERIFICATION",
    organizationId: organization.id,
    userRoles: {
      create: {
        roleId: role.id,
      },
    },
  },
});
```

<Info>
  New users are created with `PENDING_VERIFICATION` status and require admin approval before they can log in.
</Info>

## Authorization Helpers

### Require Authentication

```typescript
import { requireAuth } from "@/lib/api-helpers";

export async function GET(req: NextRequest) {
  const authResult = await requireAuth();
  if ("error" in authResult) return authResult.error;
  
  // Access session data
  const userId = authResult.session.user.id;
  const orgId = authResult.session.user.organizationId;
}
```

### Require Permission

```typescript
import { requirePermission } from "@/lib/api-helpers";

const permissionError = requirePermission(
  authResult.session.user.permissions,
  "forest-patrimony",
  "CREATE"
);
if (permissionError) return permissionError;
```

## Audit Trail

### Authentication Events

All authentication events are logged to the `AuditLog` table:

```prisma
enum AuditAction {
  LOGIN
  LOGOUT
  LOGIN_FAILED
  PASSWORD_RESET
  PERMISSION_CHANGE
  // ... other actions
}

model AuditLog {
  id             String      @id @default(uuid())
  userId         String?
  sessionId      String?
  action         AuditAction
  entityType     String?
  entityId       String?
  oldValues      Json?
  newValues      Json?
  ipAddress      String?
  userAgent      String?
  metadata       Json?
  createdAt      DateTime    @default(now())
}
```
