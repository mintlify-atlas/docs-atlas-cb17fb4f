## Overview

SMyEG provides comprehensive **geospatial data management** for Level 4 forest patrimony units, including shapefile import, geometry processing with PostGIS, and interactive map visualization. The system maintains versioned geometry data and supports complex GIS operations like split, merge, and land use change tracking.

## Geometry Storage Model

### ForestGeometryN4 Schema

```prisma
model ForestGeometryN4 {
  id             String                 @id @default(uuid())
  organizationId String                 @map("organization_id")
  level2Id       String                 @map("level2_id")
  level3Id       String                 @map("level3_id")
  level4Id       String                 @map("level4_id")
  
  // PostGIS geometry columns
  geom           Unsupported("geometry(MultiPolygon, 4326)")
  centroid       Unsupported("geometry(Point, 4326)")
  superficieHa   Decimal                @default(0) @db.Decimal(12, 4)
  
  // Temporal versioning
  validFrom      DateTime               @default(now())
  validTo        DateTime?
  isActive       Boolean                @default(true)
  
  // Import tracking
  importJobId    String?
  createdAt      DateTime               @default(now())
  updatedAt      DateTime               @updatedAt
  
  level4         ForestPatrimonyLevel4
  importJob      GeoImportJob?
  
  @@map("forest_geometry_n4")
}
```

<Note>
  Geometries use **SRID 4326** (WGS 84) coordinate reference system and are stored as **MultiPolygon** types to support complex shapes.
</Note>

### Geometry Versioning

The system maintains temporal versioning:

```sql
-- Active geometry
SELECT * FROM forest_geometry_n4
WHERE level4_id = '...' AND is_active = TRUE;

-- Historical geometries
SELECT * FROM forest_geometry_n4
WHERE level4_id = '...' AND is_active = FALSE
ORDER BY valid_to DESC;
```

**Key fields:**
- `validFrom`: When this geometry version became active
- `validTo`: When this geometry version was superseded (NULL for current)
- `isActive`: Only one geometry per Level 4 can be active at a time

### Import Flow

<Steps>
  <Step title="File Upload">
    User uploads shapefile (`.shp`, `.shx`, `.dbf`, `.prj`) via `/api/forest/geo/import`
  </Step>
  <Step title="Job Creation">
    System creates `GeoImportJob` with status PENDING and returns job ID
  </Step>
  <Step title="Async Processing">
    Worker endpoint `/api/forest/geo/import/worker` processes the job:
    - **EXTRACTING**: Read shapefile features
    - **VALIDATING**: Check required fields and geometry validity
    - **PROCESSING**: Insert geometries into PostGIS
  </Step>
  <Step title="Status Polling">
    Client polls `/api/forest/geo/import/[jobId]` for progress updates
  </Step>
  <Step title="Completion">
    Job status becomes COMPLETED or FAILED, geometries are active in the map
  </Step>
</Steps>

### Import API Endpoints

```typescript
// POST /api/forest/geo/import
export async function POST(req: NextRequest) {
  const formData = await req.formData();
  const file = formData.get("file") as File;
  const variationDate = formData.get("variationDate") as string;
  const variationNotes = formData.get("variationNotes") as string;
  
  // Create import job
  const job = await prisma.geoImportJob.create({
    data: {
      organizationId,
      fileName: file.name,
      storagePath: savedPath,
      status: "PENDING",
      metadata: {
        variationDate,
        variationNotes,
      },
    },
  });
  
  return ok({ jobId: job.id }, 201);
}

// GET /api/forest/geo/import/[jobId]
export async function GET(
  req: NextRequest,
  { params }: { params: { jobId: string } }
) {
  const job = await prisma.geoImportJob.findUnique({
    where: { id: params.jobId },
    include: {
      items: {
        orderBy: { featureIndex: "asc" },
        take: 100, // First 100 items for error preview
      },
    },
  });
  
  return ok({ job });
}
```

### Haversine Distance Calculation

```typescript
function haversineDistanceMeters(a: PointerCoords, b: PointerCoords) {
  const toRadians = (value: number) => (value * Math.PI) / 180;
  const earthRadiusMeters = 6371000;

  const lat1 = toRadians(a.lat);
  const lat2 = toRadians(b.lat);
  const deltaLat = toRadians(b.lat - a.lat);
  const deltaLng = toRadians(b.lng - a.lng);

  const h = Math.sin(deltaLat / 2) ** 2
    + Math.cos(lat1) * Math.cos(lat2) * Math.sin(deltaLng / 2) ** 2;

  return 2 * earthRadiusMeters * Math.asin(Math.sqrt(h));
}
```

## GeoJSON Layer API

### Get Level 4 Geometries

```typescript
// GET /api/forest/geo/layers/nivel4?bbox=-180,-85,180,85
export async function GET(req: NextRequest) {
  const bbox = req.nextUrl.searchParams.get("bbox") ?? "-180,-85,180,85";
  const [minLng, minLat, maxLng, maxLat] = bbox.split(",").map(Number);
  
  const features = await prisma.$queryRaw<RawGeometry[]>`
    SELECT
      g.id,
      g.level4_id,
      l4.code AS level4_code,
      l4.name AS level4_name,
      l3.id AS level3_id,
      l3.code AS level3_code,
      l2.id AS level2_id,
      l2.code AS level2_code,
      g.superficie_ha,
      ST_AsGeoJSON(g.geom)::json AS geometry
    FROM forest_geometry_n4 g
    JOIN forest_patrimony_level4 l4 ON g.level4_id = l4.id
    JOIN forest_patrimony_level3 l3 ON l4.level3_id = l3.id
    JOIN forest_patrimony_level2 l2 ON l3.level2_id = l2.id
    WHERE g.is_active = TRUE
      AND ST_Intersects(
        g.geom,
        ST_MakeEnvelope(${minLng}, ${minLat}, ${maxLng}, ${maxLat}, 4326)
      )
  `;
  
  const featureCollection: FeatureCollection = {
    type: "FeatureCollection",
    features: features.map(f => ({
      type: "Feature",
      geometry: f.geometry,
      properties: {
        level4Id: f.level4_id,
        level4Code: f.level4_code,
        level4Name: f.level4_name,
        surfaceHa: Number(f.superficie_ha),
      },
    })),
  };
  
  return ok({ data: featureCollection });
}
```

## PostGIS Spatial Queries

### Calculate Area in Hectares

```sql
SELECT
  id,
  level4_id,
  ST_Area(geom::geography) / 10000.0 AS superficie_ha
FROM forest_geometry_n4
WHERE is_active = TRUE;
```

### Find Geometries by Point

```sql
SELECT *
FROM forest_geometry_n4
WHERE is_active = TRUE
  AND ST_Contains(geom, ST_SetSRID(ST_MakePoint(-66.5, 8.5), 4326));
```

### Buffer Analysis

```sql
SELECT
  id,
  ST_Buffer(geom::geography, 100)::geometry AS buffered_geom
FROM forest_geometry_n4
WHERE level4_id = 'abc-123';
```
