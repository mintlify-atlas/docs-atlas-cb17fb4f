## Overview

SMyEG provides comprehensive **analytics and audit logging** capabilities for monitoring system usage, tracking data changes, and generating insights from forest management operations. The system includes dashboard metrics, detailed audit trails, and export functionality.

## Dashboard Metrics

### Metrics API

```typescript
// GET /api/dashboard/metrics
export async function GET() {
  const authResult = await requireAuth();
  if ("error" in authResult) return authResult.error;

  const [usersTotal, usersActive, pendingNotifications, auditTotal] = 
    await Promise.all([
      prisma.user.count({ where: { deletedAt: null } }),
      prisma.user.count({ where: { status: "ACTIVE", deletedAt: null } }),
      prisma.notification.count({ where: { status: "PENDING" } }),
      prisma.auditLog.count(),
    ]);

  return ok({
    usersTotal,
    usersActive,
    pendingNotifications,
    auditTotal,
    generatedAt: new Date().toISOString(),
  });
}
```

## Audit Logging System

### Audit Log Model

```prisma
model AuditLog {
  id             String      @id @default(uuid())
  organizationId String?
  userId         String?
  sessionId      String?
  action         AuditAction
  entityType     String?     // e.g., "User", "ForestPatrimonyLevel4"
  entityId       String?
  entityName     String?
  oldValues      Json?       // Previous state (for UPDATE)
  newValues      Json?       // New state (for CREATE/UPDATE)
  changedFields  String[]    // List of modified fields

  userAgent      String?
  requestId      String?     // Trace request across services
  durationMs     Int?        // Operation duration
  statusCode     Int?        // HTTP status code
  errorMessage   String?
  metadata       Json?       // Additional context
  createdAt      DateTime    @default(now())
  
  organization   Organization?
  user           User?
  session        Session?
}

enum AuditAction {
  CREATE
  UPDATE
  DELETE
  LOGIN
  LOGOUT
  LOGIN_FAILED
  PASSWORD_RESET
  PERMISSION_CHANGE
  EXPORT
  VIEW
}
```

## Audit Log API

### Query Audit Logs

```typescript
// GET /api/audit?page=1&limit=50&action=UPDATE&entityType=User
export async function GET(req: NextRequest) {
  const authResult = await requireAuth();
  if ("error" in authResult) return authResult.error;

  const permissionError = requirePermission(
    authResult.session.user.permissions,
    "audit",
    "READ"
  );
  if (permissionError) return permissionError;

  const page = Number(req.nextUrl.searchParams.get("page") ?? 1);
  const limit = Number(req.nextUrl.searchParams.get("limit") ?? 50);
  const action = req.nextUrl.searchParams.get("action") ?? undefined;
  const entityType = req.nextUrl.searchParams.get("entityType") ?? undefined;

  const where = {
    ...(action ? { action: action as AuditAction } : {}),
    ...(entityType ? { entityType } : {}),
  };

  const [total, logs] = await Promise.all([
    prisma.auditLog.count({ where }),
    prisma.auditLog.findMany({
      where,
      skip: (page - 1) * limit,
      take: limit,
      orderBy: { createdAt: "desc" },
      include: {
        user: {
          select: {
            email: true,
            firstName: true,
            lastName: true,
          },
        },
      },
    }),
  ]);

  return ok({
    items: logs,
    pagination: {
      total,
      page,
      limit,
      totalPages: Math.ceil(total / limit),
    },
  });
}
```

### Export Audit Logs

```typescript
// POST /api/audit (Export)
export async function POST(req: NextRequest) {
  const authResult = await requireAuth();
  if ("error" in authResult) return authResult.error;

  const permissionError = requirePermission(
    authResult.session.user.permissions,
    "audit",
    "EXPORT"
  );
  if (permissionError) return permissionError;

  const body = await req.json();
  const exportFormat = body?.format === "json" ? "json" : "csv";

  // Log the export action
  await prisma.auditLog.create({
    data: {
      userId: authResult.session.user.id,
      action: "EXPORT",
      entityType: "AuditLog",
      metadata: { format: exportFormat },
    },
  });

  const logs = await prisma.auditLog.findMany({
    orderBy: { createdAt: "desc" },
    take: 500, // Limit to 500 most recent
  });

  if (exportFormat === "json") {
    return ok({ 
      checksum: `audit-${Date.now()}`, 
      data: logs 
    });
  }

  // CSV export
  const header = "id,action,entityType,entityId,userId,createdAt";
  const rows = logs.map((log) =>
    [
      log.id,
      log.action,
      log.entityType ?? "",
      log.entityId ?? "",
      log.userId ?? "",
      log.createdAt.toISOString(),
    ].join(",")
  );

  return ok({ 
    checksum: `audit-${Date.now()}`, 
    csv: [header, ...rows].join("\n") 
  });
}
```

### Delete Audit Logs

```typescript
// DELETE /api/audit (Admin only)
export async function DELETE(req: NextRequest) {
  const authResult = await requireAuth();
  if ("error" in authResult) return authResult.error;

  const permissionError = requirePermission(
    authResult.session.user.permissions,
    "audit",
    "DELETE"
  );
  if (permissionError) return permissionError;

  try {
    await prisma.auditLog.deleteMany({});
    return ok({ success: true });
  } catch (error) {
    return fail("No se pudieron eliminar los registros de auditoría", 500);
  }
}
```

### Profile API

```typescript
// GET /api/profile
export async function GET(req: NextRequest) {
  const authResult = await requireAuth();
  if ("error" in authResult) return authResult.error;

  const user = await prisma.user.findUnique({
    where: { id: authResult.session.user.id },
    include: {
      profile: true,
      organization: {
        select: { id: true, name: true, logoUrl: true },
      },
      userRoles: {
        include: {
          role: {
            select: { name: true, slug: true },
          },
        },
      },
    },
  });

  return ok({ user });
}

// PATCH /api/profile
export async function PATCH(req: NextRequest) {
  const authResult = await requireAuth();
  if ("error" in authResult) return authResult.error;

  const body = await req.json();
  const parsed = updateProfileSchema.safeParse(body);
  if (!parsed.success) {
    return fail("Datos inválidos", 400, parsed.error.flatten());
  }

  const updated = await prisma.user.update({
    where: { id: authResult.session.user.id },
    data: {
      firstName: parsed.data.firstName,
      lastName: parsed.data.lastName,
      displayName: parsed.data.displayName,
      profile: {
        upsert: {
          create: parsed.data.profile,
          update: parsed.data.profile,
        },
      },
    },
  });

  return ok({ user: updated });
}
```
