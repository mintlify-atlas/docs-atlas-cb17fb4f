## Overview

SMyEG provides a comprehensive **asset measurement system** for tracking biological, hydrological, dendrometric, and demographic data through structured measurement campaigns. The system supports multiple measurement protocols, flexible data schemas, and automated calculation pipelines.

## Measurement Campaigns

### Campaign Model

```prisma
model AssetMeasurementCampaign {
  id                String                  @id @default(uuid())
  organizationId    String
  level4Id          String                  // Patrimony unit
  domain            AssetDomain
  protocolType      MeasurementProtocolType
  campaignCode      String                  // Unique identifier
  campaignName      String
  campaignDate      DateTime
  schemaVersion     String                  @default("1.0")
  status            AssetCampaignStatus     @default(DRAFT)
  notes             String?
  metadata          Json?                   // Custom configuration
  createdById       String?
  createdAt         DateTime                @default(now())
  updatedAt         DateTime                @updatedAt
  
  organization      Organization
  level4            ForestPatrimonyLevel4
  createdBy         User?
  observations      AssetMeasurementObservation[]
  results           AssetResultSnapshot[]
  calculationJobs   AssetCalculationJob[]
  
  @@unique([organizationId, campaignCode])
  @@map("asset_measurement_campaigns")
}

enum AssetCampaignStatus {
  DRAFT       // Being configured
  ACTIVE      // Collecting observations
  CLOSED      // No more data entry
  CANCELLED   // Cancelled
}
```

### Bulk Import Observations

```typescript
// POST /api/assets/measurement-observations/import
export async function POST(req: NextRequest) {
  const body = await req.json();
  const parsed = importMeasurementObservationsSchema.safeParse(body);
  
  if (!parsed.success) {
    return fail("Datos inválidos", 400, parsed.error.flatten());
  }
  
  const campaign = await prisma.assetMeasurementCampaign.findFirst({
    where: {
      id: parsed.data.campaignId,
      organizationId,
    },
  });
  
  if (!campaign) {
    return fail("Campaña no encontrada", 404);
  }
  
  // Bulk insert observations
  const created = await prisma.assetMeasurementObservation.createMany({
    data: parsed.data.observations.map(obs => ({
      campaignId: parsed.data.campaignId,
      level5UnitCode: obs.level5UnitCode,
      sectionKey: obs.sectionKey,
      observedAt: obs.observedAt,
      payload: obs.payload as Prisma.InputJsonValue,
    })),
  });
  
  return ok({ count: created.count }, 201);
}
```

## Result Snapshots

### Result Model

```prisma
model AssetResultSnapshot {
  id                String                  @id @default(uuid())
  campaignId        String
  level4Id          String
  domain            AssetDomain
  resultKey         String                  // e.g., "survival_summary"
  resultVersion     Int                     @default(1)
  schemaVersion     String                  @default("1.0")
  resultPayload     Json                    // Calculated metrics
  isCurrent         Boolean                 @default(true)
  calculatedAt      DateTime                @default(now())
  createdAt         DateTime                @default(now())
  updatedAt         DateTime                @updatedAt
  
  campaign          AssetMeasurementCampaign
  level4            ForestPatrimonyLevel4
  
  @@unique([campaignId, resultKey, resultVersion])
  @@index([level4Id, domain, isCurrent, calculatedAt])
  @@map("asset_result_snapshots")
}
```

### Result Versioning

Results are versioned to track recalculations:

- **resultVersion**: Increments with each recalculation
- **isCurrent**: Only the latest version has `isCurrent = true`
- **calculatedAt**: Timestamp of calculation

**Example result payload:**

```json
{
  "resultKey": "survival_summary",
  "resultVersion": 2,
  "resultPayload": {
    "totalPlots": 25,
    "totalPlanted": 6250,
    "totalSurviving": 5912,
    "totalDead": 338,
    "overallSurvivalRate": 94.6,
    "survivalRateByPlot": [
      { "plotCode": "PLOT-001", "rate": 94.8 },
      { "plotCode": "PLOT-002", "rate": 93.2 }
    ],
    "mortalityCauses": {
      "Drought": 198,
      "Disease": 85,
      "Pest": 55
    }
  },
  "calculatedAt": "2024-01-15T16:30:00Z"
}
```

## Calculation Jobs

### Job Model

```prisma
model AssetCalculationJob {
  id                String                  @id @default(uuid())
  organizationId    String
  campaignId        String
  domain            AssetDomain
  protocolType      MeasurementProtocolType
  status            AssetCalculationStatus  @default(PENDING)
  attempts          Int                     @default(0)
  runAfter          DateTime                @default(now())
  startedAt         DateTime?
  completedAt       DateTime?
  errorMessage      String?
  payload           Json?                   // Calculation parameters
  createdById       String?
  createdAt         DateTime                @default(now())
  updatedAt         DateTime                @updatedAt
  
  organization      Organization
  campaign          AssetMeasurementCampaign
  createdBy         User?
  
  @@map("asset_calculation_jobs")
}

enum AssetCalculationStatus {
  PENDING
  PROCESSING
  COMPLETED
  FAILED
}
```

### Job Queue Processing

```typescript
// Worker endpoint: POST /api/assets/calculation-jobs/worker
export async function POST(req: NextRequest) {
  // Fetch pending jobs
  const jobs = await prisma.assetCalculationJob.findMany({
    where: {
      status: "PENDING",
      runAfter: { lte: new Date() },
      attempts: { lt: 3 },
    },
    orderBy: { createdAt: "asc" },
    take: 10,
  });
  
  for (const job of jobs) {
    try {
      // Mark as processing
      await prisma.assetCalculationJob.update({
        where: { id: job.id },
        data: {
          status: "PROCESSING",
          startedAt: new Date(),
          attempts: { increment: 1 },
        },
      });
      
      // Fetch observations
      const observations = await prisma.assetMeasurementObservation.findMany({
        where: { campaignId: job.campaignId },
      });
      
      // Run calculation based on protocol
      const results = await calculateMetrics(
        job.protocolType,
        observations
      );
      
      // Save results
      await prisma.assetResultSnapshot.create({
        data: {
          campaignId: job.campaignId,
          level4Id: campaign.level4Id,
          domain: job.domain,
          resultKey: results.key,
          resultPayload: results.payload,
          isCurrent: true,
        },
      });
      
      // Mark job as completed
      await prisma.assetCalculationJob.update({
        where: { id: job.id },
        data: {
          status: "COMPLETED",
          completedAt: new Date(),
        },
      });
    } catch (error) {
      // Handle failure
      await prisma.assetCalculationJob.update({
        where: { id: job.id },
        data: {
          status: job.attempts >= 2 ? "FAILED" : "PENDING",
          errorMessage: error.message,
          runAfter: new Date(Date.now() + 5 * 60 * 1000), // Retry in 5 min
        },
      });
    }
  }
  
  return ok({ processedJobs: jobs.length });
}
```

## API Endpoints

### Campaign Management

| Endpoint | Method | Description |
|----------|--------|-------------|
| `/api/assets/measurement-campaigns` | GET | List campaigns with filters |
| `/api/assets/measurement-campaigns` | POST | Create new campaign |
| `/api/assets/measurement-campaigns` | PATCH | Update campaign |
| `/api/assets/measurement-campaigns` | DELETE | Delete campaign (if no data) |

**Query parameters:**
- `level4Id` - Filter by patrimony unit
- `domain` - Filter by measurement domain
- `protocolType` - Filter by protocol
- `status` - Filter by campaign status
- `search` - Search by code or name
- `page`, `limit` - Pagination
- `sortBy`, `sortOrder` - Sorting

### Observation Management

| Endpoint | Method | Description |
|----------|--------|-------------|
| `/api/assets/measurement-observations` | GET | List observations for campaign |
| `/api/assets/measurement-observations` | POST | Create single observation |
| `/api/assets/measurement-observations/import` | POST | Bulk import observations |
| `/api/assets/measurement-observations/export` | GET | Export observations (CSV/JSON) |

### Results & Calculations

| Endpoint | Method | Description |
|----------|--------|-------------|
| `/api/assets/result-snapshots` | GET | Query result snapshots |
| `/api/assets/calculation-jobs` | POST | Manually trigger calculation |
| `/api/assets/calculation-jobs/worker` | POST | Worker endpoint (cron) |
